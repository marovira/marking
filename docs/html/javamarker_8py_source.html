147  def performDiff(self, expected, ans):
148  """
149  Performs the diff between the student's output and the master output.
150 
151  Parameters:
152  ----------
153  expected:
154  The master output to compare against.
155  ans:
156  The students answer.
157 
158  Returns:
159  -------
160  0 if the diff fails, 1 otherwise. It will also return the results of
161  the diff.
162 
163  """
164  if len(ans) == 0:
165  return 0, []
166 
167  d = difflib.Differ()
168  diff = list(d.compare(expected, ans))
169  if len(diff) != len(expected):
170  return 0, diff
171  for line in diff:
172  if re.search('(^[+] .*)|^(- ).*|^([?].*)', line):
173  return 0, diff
174  return 1, []
